41 public List<LocalDateTime> obtenerHorasDisponibles(Long idMedico, LocalDate fecha) {
42 if(fecha.getDayOfWeek().equals(DayOfWeek.SATURDAY) || fecha.getDayOfWeek().equals(DayOfWeek.SUNDAY) || fecha.isBefore(LocalDate.now())) {
43 return new ArrayList<>();
44 }
45
46 LocalDateTime horaActual = LocalDateTime.now();
47 List<Cita> citasPlaneadas = obtenerCitasPorDia(idMedico, fecha);
48
49 ArrayList<LocalDateTime> horasDisponibles = new ArrayList<>();
50 for(int hora = CITAS_HORA_INICIO; hora < CITAS_HORA_FIN; hora++) {
51 for(int numeroCita = 0; numeroCita < CITAS_POR_HORA; numeroCita++) {
52 int minutos = 60 / CITAS_POR_HORA * numeroCita;
53 LocalDateTime horaDeseada = LocalDateTime.of(fecha, LocalTime.of(hora, minutos));
54
55 if(horaDeseada.isBefore(horaActual) || horaDeseada.isEqual(horaActual)) {
56 continue;
57 }
58
59
60 boolean horaOcupada = citasPlaneadas.stream().anyMatch((citaPlaneada) -> horaDeseada.equals(citaPlaneada.getFecha()));
61 if(horaOcupada) {
62 continue;
63 }
64
65 horasDisponibles.add(horaDeseada);
66 }
67 }
68
69 return horasDisponibles;
70 }
