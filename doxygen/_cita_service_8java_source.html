40 public List<LocalDateTime> obtenerHorasDisponibles(Long idMedico, LocalDate fecha) {
41 if(fecha.getDayOfWeek().equals(DayOfWeek.SATURDAY) || fecha.getDayOfWeek().equals(DayOfWeek.SUNDAY) || fecha.isBefore(LocalDate.now())) {
42 return new ArrayList<>();
43 }
44
45 LocalDateTime horaActual = LocalDateTime.now();
46 List<Cita> citasPlaneadas = obtenerCitasPorDia(idMedico, fecha);
47
48 ArrayList<LocalDateTime> horasDisponibles = new ArrayList<>();
49 for(int hora = CITAS_HORA_INICIO; hora < CITAS_HORA_FIN; hora++) {
50 for(int numeroCita = 0; numeroCita < CITAS_POR_HORA; numeroCita++) {
51 int minutos = 60 / CITAS_POR_HORA * numeroCita;
52 LocalDateTime horaDeseada = LocalDateTime.of(fecha, LocalTime.of(hora, minutos));
53
54 if(horaDeseada.isBefore(horaActual) || horaDeseada.isEqual(horaActual)) {
55 continue;
56 }
57
58
59 boolean horaOcupada = citasPlaneadas.stream().anyMatch((citaPlaneada) -> horaDeseada.equals(citaPlaneada.getFecha()));
60 if(horaOcupada) {
61 continue;
62 }
63
64 horasDisponibles.add(horaDeseada);
65 }
66 }
67
68 return horasDisponibles;
69 }
